* Sanity & visual basics

Quiet startup, no GUI chrome, generous font, sensible splits, soft line wrapping.

#+BEGIN_SRC emacs-lisp
  (setq inhibit-startup-screen t
        visible-bell           1
        ring-bell-function     'ignore
        use-dialog-box         nil
        use-short-answers      t)        ; y/n instead of yes/no, Emacs 28+

  ;; Modern line numbers (display-line-numbers-mode replaces linum)
  (global-display-line-numbers-mode 1)

  ;; Soft line wrap
  (global-visual-line-mode 1)

  ;; Splits prefer side-by-side
  (setq split-height-threshold nil
        split-width-threshold  100)

  ;; Font (only set when running graphical Emacs)
  (when (display-graphic-p)
    (set-face-attribute 'default nil :family "JetBrains Mono" :height 160))

  ;; Strip GUI chrome
  (when (fboundp 'tool-bar-mode)   (tool-bar-mode -1))
  (when (fboundp 'menu-bar-mode)   (menu-bar-mode -1))
  (when (fboundp 'scroll-bar-mode) (scroll-bar-mode -1))
  (when (fboundp 'fringe-mode)     (fringe-mode 8))

  ;; Pixel-precise smooth scrolling (Emacs 29+)
  (when (fboundp 'pixel-scroll-precision-mode)
    (pixel-scroll-precision-mode 1))

  ;; Frame transparency to match WezTerm. alpha-background is Emacs 29+.
  (when (boundp 'default-frame-alist)
    (add-to-list 'default-frame-alist '(alpha-background . 92)))

  ;; Modeline clock
  (display-time-mode 1)
#+END_SRC

* Minibuffer completion: vertico + consult + marginalia + orderless

Modern selectrum-family stack, replaces ivy/swiper/counsel.

#+BEGIN_SRC emacs-lisp
  (use-package vertico
    :demand t
    :init (vertico-mode 1)
    :config (setq vertico-cycle t
                  vertico-resize t))

  (use-package marginalia
    :demand t
    :init (marginalia-mode 1))

  (use-package orderless
    :demand t
    :config
    (setq completion-styles '(orderless basic)
          completion-category-overrides '((file (styles basic partial-completion)))))

  (use-package consult
    :bind (("C-s"   . consult-line)
           ("C-x b" . consult-buffer)
           ("M-y"   . consult-yank-pop)
           ("M-g g" . consult-goto-line)
           ("M-g i" . consult-imenu)))

  (use-package nerd-icons-completion
    :after marginalia
    :config
    (nerd-icons-completion-mode 1)
    (add-hook 'marginalia-mode-hook #'nerd-icons-completion-marginalia-setup))
#+END_SRC

* In-buffer completion: corfu + cape

Replaces company. Child-frame UI, snappier.

#+BEGIN_SRC emacs-lisp
  (use-package corfu
    :demand t
    :init (global-corfu-mode 1)
    :config
    (setq corfu-cycle t
          corfu-auto t
          corfu-auto-delay 0.1
          corfu-auto-prefix 2
          corfu-quit-no-match 'separator))

  (use-package cape
    :demand t
    :init
    (add-to-list 'completion-at-point-functions #'cape-dabbrev)
    (add-to-list 'completion-at-point-functions #'cape-file))
#+END_SRC

* LSP via eglot (built-in)

Replaces elpy and works for most languages out of the box.

#+BEGIN_SRC emacs-lisp
  (use-package eglot
    :straight nil                        ; built-in since Emacs 29
    :hook ((python-mode . eglot-ensure)
           (python-ts-mode . eglot-ensure)
           (js-mode . eglot-ensure)
           (typescript-mode . eglot-ensure)
           (rust-mode . eglot-ensure)
           (go-mode . eglot-ensure))
    :config
    ;; Speed up eglot for snappier interaction.
    (setq eglot-events-buffer-size 0
          eglot-autoshutdown t))
#+END_SRC
